def generate_stopover_variants(base_fare: VerifiedFare):
    hub = base_fare.stopover_iata
    if not hub:
        return []

    depart = parse_date(base_fare.depart_date)
    ret = parse_date(base_fare.return_date)

    variants = []

    # Extend-trip mode
    for out_nights in (2, 3):
        for back_nights in (2, 3):
            segments = [
                {"origin": base_fare.origin, "dest": hub, "date": iso(depart)},
                {"origin": hub, "dest": base_fare.dest, "date": iso(depart + days(out_nights))},
                {"origin": base_fare.dest, "dest": hub, "date": iso(ret + days(out_nights))},
                {"origin": hub, "dest": base_fare.origin, "date": iso(ret + days(out_nights + back_nights))},
            ]
            variants.append({
                "mode": "extend_trip",
                "out_nights": out_nights,
                "back_nights": back_nights,
                "segments": segments,
            })

    # Fixed-trip mode
    trip_len = (ret - depart).days
    for out_nights in (2, 3):
        for back_nights in (2, 3):
            if out_nights + back_nights >= trip_len - 1:
                continue
            us_arrive = depart + days(out_nights)
            us_depart = ret - days(back_nights)
            segments = [
                {"origin": base_fare.origin, "dest": hub, "date": iso(depart)},
                {"origin": hub, "dest": base_fare.dest, "date": iso(us_arrive)},
                {"origin": base_fare.dest, "dest": hub, "date": iso(us_depart)},
                {"origin": hub, "dest": base_fare.origin, "date": iso(ret)},
            ]
            variants.append({
                "mode": "fixed_trip",
                "out_nights": out_nights,
                "back_nights": back_nights,
                "segments": segments,
            })

    return variants

def score_stopover_opportunity(base_pp, stopover_pp, total_nights, verified_grade, single_ticket_ok):
    delta = stopover_pp - base_pp
    delta_pct = delta / max(base_pp, 1)

    score = 0

    # cheaper premium is better
    if delta <= 80:
        score += 40
    elif delta <= 150:
        score += 25
    elif delta <= 250:
        score += 10

    if delta_pct <= 0.15:
        score += 25
    elif delta_pct <= 0.25:
        score += 10

    if total_nights >= 4:
        score += 20
    elif total_nights >= 2:
        score += 10

    if verified_grade == "A":
        score += 15
    elif verified_grade == "B":
        score += 8

    if single_ticket_ok:
        score += 10
    else:
        score -= 20

    return score

def classify_booking_rows(rows):
    """
    rows from the booking-options area
    """
    best_airline_direct = None
    has_separate_tickets_warning = False

    for row in rows:
        txt = row["text"].lower()
        if "separate tickets" in txt or "booked individually" in txt:
            has_separate_tickets_warning = True

        if row["partner_type"] == "airline" and not ("separate tickets" in txt):
            if best_airline_direct is None or row["price_usd"] < best_airline_direct["price_usd"]:
                best_airline_direct = row

    return {
        "best_airline_direct": best_airline_direct,
        "has_separate_tickets_warning": has_separate_tickets_warning,
    }
